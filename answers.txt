def calculate_overfishing_percentage(total_fish_stocks, overexploited_fish_stocks):
    """
    Calculate the percentage of fish stocks overexploited.
    
    Args:
        total_fish_stocks (int): Total number of fish stocks.
        overexploited_fish_stocks (int): Number of overexploited fish stocks.
    
    Returns:
        float: Percentage of overexploited fish stocks.
    """
    if total_fish_stocks == 0:
        return 0.0  # Avoid division by zero
    percentage = (overexploited_fish_stocks / total_fish_stocks) * 100
    return round(percentage, 2)
